Nexus Dashboard

REACT FIREBASE STATE MANAGEMENT

Nexus is a Task Management System concept. It uses a real-time database to sync tasks across different users. The goal was to create a clutter-free UI that helps small teams focus on their most urgent work.

App.js (React Component) 
import React, { useState, useEffect } from 'react';
import { db } from './firebase-config';

// Main Dashboard Component
function NexusDashboard() {
    const [tasks, setTasks] = useState([]);
    const [input, setInput] = useState('');

    // Add task to state and cloud
    const addTask = (e) => {
        e.preventDefault();
        setTasks([...tasks, { text: input, completed: false }]);
        setInput('');
    };

    return (
        <div className="dashboard-container">
            <h1>Nexus Tasks</h1>
            <form>
                <input
                    value={input}
                    onChange={e => setInput(e.target.value)}
                    placeholder="Enter a new task..."
                />
                <button onClick={addTask}>Add Task</button>
            </form>

            <ul>
                {tasks.map(task => (
                    <li>{task.text}</li>
                ))}
            </ul>
        </div>
    );
}

export default NexusDashboard;
