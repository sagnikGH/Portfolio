Velo Delivery API

SPRING BOOT REST API BACKEND

Velo is a high-performance backend system designed to handle logistics and courier tracking. It demonstrates how Object-Oriented Programming (OOP) can be applied to manage real-world data like delivery status and driver coordinates.

DeliveryController.java 
package com.velo.api;

import org.springframework.web.bind.annotation.*;
import java.util.List;

@RestController
@RequestMapping("/api/delivery")
public class DeliveryController {

    // API endpoint to get delivery status by ID
    @GetMapping("/{id}")
    public Delivery getDeliveryStatus(@PathVariable String id) {
        // ICSE Logic: Objects and Classes in action
        Delivery d = new Delivery();
        d.setId(id);
        d.setStatus("IN_TRANSIT");
        d.setEstimatedArrival("15 mins");

        return d;
    }

    @PostMapping("/update")
    public String updateLocation(@RequestBody Location loc) {
        // Logic for Geofencing would go here
        System.out.println("Driver at: " + loc.getLat() + ", " + loc.getLng());
        return "Location updated successfully!";
    }
}
